hostname KNYV-SW
no ip domain-lookup
ip domain-name horizonbank.hu
banner motd # Az illetektelen hozzaferes TILOS! #
security passwords min-length 8
clock timezone CET 1
service password-encryption
enable secret K0nyv3lesPR1v1

exit
